89 /* I2C Master Controller Status
90  * bit 0 = controller busy: all other status bits invalid
91  * bit 1 = error condition
92  * bit 2 = slave address was not acknowledged during last operation
93  * bit 3 = data not acknowledged during last operation
94  * bit 4 = arbitration lost during last operation
95  * bit 5 = controller idle
96  * bit 6 = bus busy
97  */
98 #define I2CM_CONTROLLER_BUSY(status) (((status) & 0x01) != 0)
99 #define I2CM_DATA_NACK(status) (((status) & 0x0A) != 0)
100 #define I2CM_ADDRESS_NACK(status) (((status) & 0x06) != 0)
101 #define I2CM_ARB_LOST(status) (((status) & 0x12) != 0)
102 #define I2CM_IDLE(status) (((status) & 0x20) != 0)
103 #define I2CM_BUS_BUSY(status) (((status) & 0x40) != 0)
